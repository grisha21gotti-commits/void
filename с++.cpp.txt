#include <iostream>
#include <stack>
using namespace std;

int main() {
    stack<string> st;   // пустой стек строк

    // Кладём элементы в стек
    st.push("9");
    st.push("99");
    st.push("999");

    // Выводим стек сверху вниз
    while (!st.empty()) {
        cout << st.top() << endl; // берём верхний
        st.pop();                 // убираем его
    }

    return 0;
}
